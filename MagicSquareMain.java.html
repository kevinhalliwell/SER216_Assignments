package Week6;

import java.util.Scanner;

public class MagicSquareMain {
	
	static double total = 0;
	static double totalSqRt = 0;
	static double sum = 0;
	static double errorCheck = 0;
	final static int ZERO = 0; //constant zero variable
	static Boolean boolLoop = true; //used to break out of loops
	static int count = 0; //used to fill the array
	static double rowSum = 0;
	
	/*
	* Main method
	*/
	public static void main(String[] args){
		
		//open instance of scanner
		Scanner keyboard = new Scanner (System.in);
		
		//user greeting
		System.out.println("Welcome to the Magic Square program.\n");
		System.out.println("Enter the numbers for your Magic Square.");
		System.out.println("Please remember a Magic Square must contain the proper "
				+ "amount of numbers to work properly. (i.e. n^3 needs 9 numbers).\n");
		
		//Create the array list
		//Allowing for any size of array. One dimensional to capture input
		MagicSquare square = new MagicSquare();
		
		int magicSquareArray[] = square.input();
		
		
		//Find square root and sum
		square.setSize(magicSquareArray.length);
		total = square.getSize();
		totalSqRt = square.squareRoot(total);
		sum = square.sum(total, totalSqRt);
		errorCheck = square.errorCheck(total,  totalSqRt);
		//System.out.println(errorCheck); //used for testing
		
		//Begin error checking
		//Check to see if the correct amount of values were entered
		if (errorCheck > ZERO){
			count++;
		}
		
		//Check for duplicate values
		if (!square.duplicateValues(magicSquareArray)){
			count++;
		}
		
		
		//Check first diagonals
		rowSum = ZERO;
		if (count > ZERO )
			System.out.println("your magic square failed");
		else{
			rowSum = square.firstDiagonal(magicSquareArray);
			if (sum == rowSum){
				System.out.println("first diag is good"); //used during testing
				count = 0;
			}
			else{
				count++;
				System.out.println("first diag failed"); //used during testing
			}
		}
		
		//reset loop parameters
		rowSum = ZERO;
		if (count > ZERO )
			System.out.println("Hey your magic square failed");
		else{
			//check second diagonals
			rowSum = square.secondDiagonal(magicSquareArray);
			if (sum == rowSum){
				boolLoop = false;
				System.out.println("second diag is good"); //used during testing
			}
			else{
				boolLoop = false;
				count++; //used to break and discontinue looping
				System.out.println("second diag failed"); //used during testing
			}
		}
		
		//reset loop parameters
		rowSum = ZERO;
		if (count > ZERO )
			System.out.println("your magic square failed");
		else{
			//check rows
			rowSum = square.rows(magicSquareArray);
			if (sum == rowSum){
				System.out.println("first row is good"); //used during testing
			}
			else{
				count++; //used to break and discontinue looping
				System.out.println("first row failed"); //used during testing
			}
		}
		
		//reset loop parameters
		if (count > ZERO )
			System.out.println("Your magic square failed");
		else{
			//check columns
			rowSum = square.columns(magicSquareArray);
			if (sum == rowSum){
				System.out.println("first column is good"); //used during testing
			}
			else{
				count++; //used to break and discontinue looping
				System.out.println("first column failed"); //used during testing
			}
		}		
		
		if (count == 0)
			System.out.println("True");
		else
			System.out.println("False");
		
		keyboard.close();
	}

}
